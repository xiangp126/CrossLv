brew cask install osxfuse

cd
mkdir rdisk
sshfs root@10.123.16.46: rdisk/

df -h
root@10.123.16.46:       1.1T  264G  849G  24% /root/rdisk

- unmount | umount
diskutil unmount ~/rdisk
umount ~/rdisk
